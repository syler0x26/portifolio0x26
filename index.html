class LearningSystem:
    def __init__(self):
        self.professor = "Wesley Syler"
        self.cursos = ["Inteligencia Artificial", "Python"]

    def iniciar(self):
        print("Inicializando ambiente educacional...")
        print(f"Professor responsavel: {self.professor}")

        for curso in self.cursos:
            print(f"Carregando curso: {curso}")

        print("Status: cursos em breve")


sistema = LearningSystem()
sistema.iniciar()
